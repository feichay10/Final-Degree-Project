  \else
    \@restonecolfalse
  \fi
  \chapter*{\listfigurename
    \@mkboth{\MakeUppercase\listfigurename}%

    \let\oldnumberline\numberline
    \renewcommand{\numberline}[1]{##1.\hspace{0.5em}}%
    \@starttoc{lof}%
  \endgroup
  \if@restonecol\twocolumn\fi
}
